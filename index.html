packages = ["pandas", "pytrends", "matplotlib", "seaborn", "datetime"] import pandas as pd from pytrends.request import TrendReq import matplotlib.pyplot as plt import seaborn as sns from datetime import date pytrends = TrendReq() pytrends.build_payload(kw_list=['chatGPT'], timeframe='2022-11-01 {}'.format(date.today())) df = pytrends.interest_over_time() df = df.reset_index() print(df.head(10)) fig1 = sns.relplot(data=df, x="date", y="chatGPT", kind="line") display(fig1, target="fig1")
